""" 1
Математическое округление
Напишите программу, которая принимает число с плавающей точкой и округляет его до целого числа
в соответствии с правилами школьной математики, а не банковского округления.
Программа должна корректно работать как с положительными, так и с отрицательными числами.
Пример вывода:
Введите вещественное число: 2.5
Округленное значение: 3
"""
import math as m
num = float(input("Введите вещественное число: "))
x = num % 1
if 0.5 <= x:
    print("Округленное значение: ", m.ceil(num))
else:
    print("Округленное значение: ", m.floor(num))
"""
Второй вариант    
"""
num = float(input("Введите вещественное число: "))
print("Округленное значение:", round(num))


""" 2
Гипотенуза треугольника
Напишите программу, которая запрашивает у пользователя длины двух катетов прямоугольного треугольника 
и вычисляет длину гипотенузы. 
Гипотенуза равна квадратному корню из суммы квадратов катетов.
Пример вывода:
Введите длину первого катета: 3
Введите длину второго катета: 4
Длина гипотенузы: 5.0
"""
a1 = int(input("Введите длину первого катета: "))
b1 = int(input("Введите длину второго катета: "))
print("Длина гипотенузы: ", m.sqrt(a1 ** 2 + b1 ** 2))
